# 21. 合并两个有序链表（简单）

标签（空格分隔）： LeetCode_HOT_100

---
将两个有序链表合并为一个新的有序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 

示例：

    输入：1->2->4, 1->3->4
    输出：1->1->2->3->4->4

### 我的做法：   
``` C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* pre = new ListNode(-1);
        ListNode* pre1 = pre;
        while(l1 != NULL && l2 != NULL){
            if(l1->val <= l2->val){
                pre->next = l1;
                l1 = l1->next;
            }else{
                pre->next = l2;
                l2 = l2->next;
            }
            pre = pre->next;
            pre->next = NULL;
        }
        if(l1 != NULL){
            pre->next = l1;
        }
        if(l2 != NULL){
            pre->next = l2;
        }
        return pre1->next;
    }
};
```
